dashboard/index.html — WebSocket Client

function getWebSocketUrl() {
  var scheme = window.location.protocol === 'https:' ? 'wss' : 'ws';
  return scheme + '://' + window.location.hostname + ':3000';
}

ws = new WebSocket(getWebSocketUrl());

ws.onmessage = function(event) {
  var data = JSON.parse(event.data);
  updateTable(data);
};

contiki_client/run-contiki.sh — Native Build + Run

# Build Contiki native MQTT client
CCACHE_DISABLE=1 make -C "${EXAMPLE_DIR}" TARGET=native clean all
